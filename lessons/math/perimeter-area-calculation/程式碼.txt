    <script>
        // --- 核心分頁機制 ---
        function switchPage(pageId) {
            document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
            document.querySelectorAll('.nav-btn').forEach(b => b.classList.remove('active'));
            document.getElementById('page-' + pageId).classList.add('active');
            
            const btns = document.querySelectorAll('.nav-btn');
            for(let i=0; i<btns.length; i++){
                if(btns[i].getAttribute('onclick').includes(pageId)){
                    btns[i].classList.add('active');
                }
            }
            if(pageId === 'm1') initM1();
            if(pageId === 'm3') drawM3('normal');
        }

        // ==================== 模組一：周長面積動畫 ====================
        const canvas1 = document.getElementById('canvas-m1');
        const ctx1 = canvas1.getContext('2d');
        let m1Scale = 40; 
        let rectX = 100, rectY = 80, rectW = 5 * m1Scale, rectH = 3 * m1Scale;
        let animationId = null;

        function initM1() {
            if(animationId) cancelAnimationFrame(animationId);
            ctx1.clearRect(0, 0, canvas1.width, canvas1.height);
            drawGrid(ctx1, canvas1.width, canvas1.height, m1Scale);
            ctx1.fillStyle = '#FFFFFF';
            ctx1.fillRect(rectX, rectY, rectW, rectH);
            ctx1.strokeStyle = '#37474F';
            ctx1.lineWidth = 4;
            ctx1.strokeRect(rectX, rectY, rectW, rectH);
            document.getElementById('m1-result').innerHTML = "點擊上方按鈕觀察動畫喔！";
            document.getElementById('m1-result').style.color = "#1565C0";
        }

        function drawGrid(ctx, w, h, scale) {
            ctx.strokeStyle = '#CFD8DC';
            ctx.lineWidth = 1;
            for(let x=0; x<=w; x+=scale) { ctx.beginPath(); ctx.moveTo(x, 0); ctx.lineTo(x, h); ctx.stroke(); }
            for(let y=0; y<=h; y+=scale) { ctx.beginPath(); ctx.moveTo(0, y); ctx.lineTo(w, y); ctx.stroke(); }
        }

        function animateM1(type) {
            if(animationId) cancelAnimationFrame(animationId);
            initM1();
            if (type === 'perimeter') {
                let progress = 0; let totalLength = 16;

        // 產生隨機題目陣列的函數
        function generateAdvQuestions() {
            advQuestions = [];

            // 輔助函數：隨機整數
            function randomInt(min, max) { return Math.floor(Math.random() * (max - min + 1)) + min; }

            // 輔助函數：建立選項並洗牌
            function makeQuestion(qText, correctTxt, wrong1Txt, wrong2Txt, correctMsg, wrongMsg) {
                let options = [correctTxt, wrong1Txt, wrong2Txt];
                // 洗牌演算法 (Fisher-Yates)
                for (let i = options.length - 1; i > 0; i--) {
                    const j = Math.floor(Math.random() * (i + 1));
                    [options[i], options[j]] = [options[j], options[i]];
                }
                let ansIdx = options.indexOf(correctTxt);
                let displayOpts = options.map((opt, i) => String.fromCharCode(65 + i) + ". " + opt);
                return { q: qText, opts: displayOpts, ans: ansIdx, correctMsg: correctMsg, wrongMsg: wrongMsg };
            }

            // Q1. 面積推周長
            let s1 = randomInt(5, 10); let a1 = s1 * s1; let p1 = s1 * 4;
            advQuestions.push(makeQuestion(
                `👑 挑戰 1 (已知面積推算周長)：<br>一個正方形的面積是 ${a1} 平方公分，請問它的「周長」是多少公分？`,
                `${p1} 公分`, `${a1} 公分`, `${s1 * 2} 公分`,
                `正確算法：邊長是 ${s1} (因為 ${s1}×${s1}=${a1})，周長 = ${s1} × 4 = ${p1} 公分。`,
                `思考方向：面積 = 邊長 × 邊長。哪兩個一樣的數字相乘會是 ${a1} 呢？找出邊長後，再算周長喔！`
            ));

            // Q2. 複合圖形挖空
            let l2 = randomInt(12, 16); let w2 = randomInt(8, 11); let s2 = randomInt(3, 5);
            let a2All = l2 * w2; let a2Pool = s2 * s2; let a2Left = a2All - a2Pool;
            advQuestions.push(makeQuestion(
                `👑 挑戰 2 (複合圖形面積挖空)：<br>長 ${l2} 公尺、寬 ${w2} 公尺的長方形土地，挖掉一個邊長 ${s2} 公尺的正方形水池。剩下的面積是多少平方公尺？`,
                `${a2Left}`, `${a2All}`, `${a2All - s2}`,
                `正確算法：全部面積 ${l2}×${w2}=${a2All}，水池面積 ${s2}×${s2}=${a2Pool}，${a2All} - ${a2Pool} = ${a2Left} 平方公尺。`,
                `思考方向：先算出全部長方形的面積，再扣掉中間水池的面積喔！`
            ));

            // Q3. 周長推面積
            let s3 = randomInt(6, 12); let p3 = s3 * 4; let a3 = s3 * s3;
            advQuestions.push(makeQuestion(
                `👑 挑戰 3 (已知周長推算面積)：<br>用一條長 ${p3} 公分的鐵絲，剛好圍成一個正方形。正方形的「面積」是多少平方公分？`,
                `${a3}`, `${p3}`, `${a3 * 10}`,
                `正確算法：邊長 = ${p3} ÷ 4 = ${s3}，面積 = ${s3} × ${s3} = ${a3} 平方公分。`,
                `思考方向：${p3} 公分是「周長」。正方形有四個一樣長的邊，先算出邊長是多少，再來算面積！`
            ));

            // Q4. 周長與長，求面積
            let w4 = randomInt(3, 6); let l4 = w4 + randomInt(2, 5); let p4 = (l4 + w4) * 2; let a4 = l4 * w4;
            advQuestions.push(makeQuestion(
                `👑 挑戰 4 (已知周長與長，求面積)：<br>長方形的周長是 ${p4} 公分，長是 ${l4} 公分，請問面積是多少平方公分？`,
                `${a4}`, `${l4 * p4}`, `${(p4 / 2) * l4}`,
                `正確算法：長+寬 = ${p4} ÷ 2 = ${p4 / 2}。寬 = ${p4 / 2} - ${l4} = ${w4}。面積 = ${l4} × ${w4} = ${a4} 平方公分。`,
                `思考方向：周長的一半是「長+寬」。先算出寬是多少，再來算面積喔！`
            ));

            // Q5. 單位換算
            let m5 = randomInt(20, 50) * 10;
            advQuestions.push(makeQuestion(
                `👑 挑戰 5 (單位換算)：<br>學校籃球場的面積大約是 ${m5} 平方公尺，換算成平方公分是多少？`,
                `${m5}0000`, `${m5}00`, `${m5}000000`,
                `正確算法：1平方公尺 = 10000平方公分，${m5} 加上四個 0，等於 ${m5}0000 平方公分。`,
                `思考方向：記得 1 平方公尺 = 10000 平方公分 (後面有4個0)，再仔細算一下！`
            ));

            // Q6. 面積大小比較
            let m6 = randomInt(2, 8); let cm6_base = randomInt(1, 9);
            let cm6 = cm6_base * 10000; let m6cm = m6 * 10000;
            let ans6, w6_1, w6_2;
            if (m6cm > cm6) { ans6 = "房間比較大"; w6_1 = "陽台比較大"; w6_2 = "一樣大"; }
            else if (m6cm < cm6) { ans6 = "陽台比較大"; w6_1 = "房間比較大"; w6_2 = "一樣大"; }

                `正確算法：${m6}平方公尺 = ${m6cm}平方公分，因為 ${Math.max(m6cm, cm6)} 比較大，所以答案是 ${ans6}。`,
                `思考方向：單位不同不能直接比大小！把 ${m6} 平方公尺換算成平方公分再來比較吧。`
            ));

            // Q7. 圖形拼湊
            let s7 = randomInt(4, 8); let p7 = (s7 * 2 + s7) * 2;
            advQuestions.push(makeQuestion(
                `👑 挑戰 7 (圖形拼湊與周長)：<br>把兩個邊長 ${s7} 公分的正方形，拼成一個大長方形。這個長方形的「周長」是多少公分？`,

                `思考方向：試著在紙上畫畫看！拼成的大長方形，長會變成多少？寬又是多少呢？再算周長！`
            ));

            // Q8. 周長與寬求長
            let w8 = randomInt(8, 12); let l8 = w8 + randomInt(5, 10); let p8 = (l8 + w8) * 2;
            advQuestions.push(makeQuestion(
                `👑 挑戰 8 (已知周長與寬，求長)：<br>將一條長 ${p8} 公分的繩子圍出一個長方形，如果寬是 ${w8} 公分，長是幾公分？`,
                `${l8}`, `${p8 - w8}`, `${(p8 / 2) + w8}`,
                `正確算法：長+寬 = ${p8} ÷ 2 = ${p8 / 2}，長 = ${p8 / 2} - ${w8} = ${l8} 公分。`,
                `思考方向：繩子長度就是周長。長+寬 = 周長 ÷ 2，想想看長應該是多少？`
            ));

            // Q9. 面積概念
            let m9 = randomInt(2, 6);
            advQuestions.push(makeQuestion(
                `👑 挑戰 9 (面積概念)：<br>邊長 ${m9} 公尺的正方形，如果用 1 平方公分的方塊去排，一排可以排幾個？`,
                `${m9 * 100} 個`, `${m9 * 10} 個`, `${m9 * 1000} 個`,
                `正確算法：1公尺 = 100公分，${m9} 公尺 = ${m9 * 100} 公分，所以一排可以排 ${m9 * 100} 個 1 公分的方塊。`,
                `思考方向：方塊的邊長是 1 公分。想想看，${m9} 公尺等於幾公分呢？`
            ));

            // Q10. 陷阱題
            let l10 = randomInt(4, 7); let w10 = randomInt(2, 3); let a10 = l10 * w10;
            advQuestions.push(makeQuestion(
                `👑 挑戰 10 (陷阱題！求指定單位面積)：<br>有一塊長方形花圃，長是 ${l10} 公尺，寬是 ${w10} 公尺，請問面積是多少「平方公分」？`,
                `${a10}0000`, `${a10}`, `${a10}000`,
                `正確算法：面積 = ${l10} × ${w10} = ${a10} 平方公尺。${a10} 平方公尺 = ${a10}0000 平方公分。`,

        // 魔王題庫答題點擊邏輯
        function handleAdvClick(qIndex, oIndex, btn) {
            const qObj = advQuestions[qIndex];
            const isCorrect = (oIndex === qObj.ans);
            const feedbackDiv = document.getElementById(`adv-f${qIndex}`);
            const parent = btn.parentElement;

            // 分數結算邏輯（只限第一次點擊計分）
            if (!questionState[qIndex].firstTryDone) {
                questionState[qIndex].firstTryDone = true;
                advAnsweredCount++;
                if (isCorrect) {
                    advScore += 10;
                    questionState[qIndex].correct = true;
                }
                updateScoreBoard();
                checkGameEnd();
            }

            // UI 更新選項顏色
            parent.querySelectorAll('.option-btn').forEach(b => {
                b.style.background = 'white';
                b.style.borderColor = '#E0E0E0';
            });

            if (isCorrect) {
                btn.style.background = '#C8E6C9'; 
                btn.style.borderColor = '#4CAF50';

                feedbackDiv.className = "feedback correct";
            } else {
                btn.style.background = '#FFCDD2'; 
                btn.style.borderColor = '#F44336';
                feedbackDiv.innerHTML = `❌ 哎呀，掉進陷阱了！<br><span style="color:#C62828; font-size: 18px; font-weight: normal; display: inline-block; margin-top: 5px;">${qObj.wrongMsg}</span>`;
                feedbackDiv.className = "feedback wrong";
                
                feedbackDiv.classList.remove('wrong');
                void feedbackDiv.offsetWidth; 
                feedbackDiv.classList.add('wrong');
            }
        }

            if (advAnsweredCount === advQuestions.length) {
                const rewardSec = document.getElementById('reward-section');
                const resultMsg = document.getElementById('adv-final-msg');
                const nameInput = document.getElementById('student-name');
                const claimBtn = document.getElementById('claim-btn');
                const retryBtn = document.getElementById('retry-btn');
                
                rewardSec.style.display = 'block';

                if (advScore >= 80) {
                    resultMsg.innerHTML = `
                        <h3 style="color: #2E7D32; font-size: 26px; margin-top: 0;">🌟 恭喜你獲得 ${advScore} 分，成功通過試煉！🌟</h3>

                    `;
                    nameInput.style.display = 'inline-block';
                    claimBtn.style.display = 'inline-flex';
                    // 為了鼓勵，過關後也可以讓他們選擇重新挑戰
                    retryBtn.style.display = 'inline-flex';
                } else {
                    resultMsg.innerHTML = `
                        <h3 style="color: #C62828; font-size: 26px; margin-top: 0;">💪 哎呀！你得了 ${advScore} 分！</h3>
                        <p style="font-size: 18px; font-weight: bold; color: #555;">需要達到 80 分才能拿到證書喔！<br>沒關係，趕快點擊下方按鈕，重新挑戰一次吧！</p>
                    `;
                    nameInput.style.display = 'none';
                    claimBtn.style.display = 'none';
                    retryBtn.style.display = 'inline-flex';
                }
                setTimeout(() => rewardSec.scrollIntoView({behavior: 'smooth'}), 300);
            }
        }
