    let currentPhase = 1, currentQuestions = [], currentIdx = 0, score = 0;

    function playSnd(id) {
        const audio = document.getElementById(id);
        if(audio) { audio.currentTime = 0; audio.play().catch(e=>{}); }
    }

    function changeMainMode(m) {
        document.getElementById('lesson-section').classList.toggle('hidden-section', m !== 'lesson');
        document.getElementById('quiz-section').classList.toggle('hidden-section', m !== 'quiz');

        const navLesson = document.getElementById('nav-lesson');
        const navQuiz = document.getElementById('nav-quiz');
        if (m === 'lesson') {
            navLesson.className = "btn-ghibli text-lg border-2 border-white";
            navQuiz.className = "btn-ghibli text-lg btn-orange opacity-80";
        } else {
            navLesson.className = "btn-ghibli text-lg opacity-80";
            navQuiz.className = "btn-ghibli text-lg btn-orange border-2 border-white";
            startPhase();
        }
    }

    function shuffle(arr) { return arr.sort(() => Math.random() - 0.5); }

    function startPhase() {
        currentIdx = 0; score = 0;
        document.getElementById('display-score').innerText = "0 分";
        document.getElementById('phase-result').classList.add('hidden-section');
        document.getElementById('question-card').classList.remove('hidden-section');
        document.getElementById('quiz-success').classList.add('hidden-section');

        if (currentPhase === 1) {
            currentQuestions = shuffle([...weightPool]);
            document.getElementById('current-phase-name').innerText = "階段一：重量挑戰";
            document.getElementById('quiz-badge').innerText = "1";
        } else {
            currentQuestions = shuffle([...areaPool]);
            document.getElementById('current-phase-name').innerText = "階段二：面積挑戰";
            document.getElementById('quiz-badge').innerText = "2";
        }
        renderQuestion();
    }

    function renderQuestion() {
        const q = currentQuestions[currentIdx];
        const opts = shuffle([...q.o]);
        document.getElementById('question-content').innerHTML = `
            <p class="text-3xl font-bold text-gray-800 mb-10">${q.q}</p>
            <div class="grid grid-cols-1 gap-5">
                ${opts.map(o => `
                    <label class="flex items-center p-6 border-4 border-gray-100 rounded-2xl cursor-pointer hover:bg-green-50 transition-all">
                        <input type="radio" name="opt" value="${o}" class="w-7 h-7 text-green-600">
                        <span class="ml-6 text-2xl font-bold">${o}</span>
                    </label>
                `).join('')}
            </div>
        `;
        document.getElementById('question-index-label').innerText = `題目 ${currentIdx + 1} / 5`;

        const actionArea = document.getElementById('quiz-action-area');
        const isLast = currentIdx === 4;
        actionArea.innerHTML = `
            <button id="next-btn" onclick="submitCurrentQuestion()" class="btn-ghibli text-xl px-10 py-4 ${isLast ? 'btn-orange' : ''}">
                ${isLast ? '完成挑戰並看成績 🏁' : '送出答案 ➜'}
            </button>
        `;

        updateProgress();
    }

    function submitCurrentQuestion() {
        const sel = document.querySelector('input[name="opt"]:checked');
        if (!sel) return;

        if (currentIdx === 4) {
            document.getElementById('quiz-action-area').innerHTML = '<span class="text-green-700 font-bold italic animate-pulse">正在計算森林評分...</span>';
        }

        const isCorrect = sel.value === currentQuestions[currentIdx].a;
        if (isCorrect) {
            score += 20;
            playSnd('snd-correct');
            setTimeout(goNext, 300);
        } else {
            playSnd('snd-wrong');
            showFeedback(currentQuestions[currentIdx]);
        }
    }

    function showFeedback(qData) {
        document.getElementById('feedback-text').innerText = qData.hint;
        document.getElementById('feedback-correct').innerText = qData.a;
        document.getElementById('feedback-modal').style.display = 'flex';
    }

    function closeFeedback() {
        document.getElementById('feedback-modal').style.display = 'none';
        goNext();
    }

    function goNext() {
        document.getElementById('display-score').innerText = `${score} 分`;
        currentIdx++;
        if (currentIdx < 5) renderQuestion();
        else handleFinishPhase();
    }

    function handleFinishPhase() {
        document.getElementById('question-card').classList.add('hidden-section');
        const res = document.getElementById('phase-result');
        const scoreDisplay = document.getElementById('result-score');
        const btnNext = document.getElementById('btn-next-phase');
        const btnCert = document.getElementById('btn-get-cert');
        const msg = document.getElementById('result-msg');
        const emoji = document.getElementById('result-emoji');

        res.classList.remove('hidden-section');
        scoreDisplay.innerText = score;

        btnNext.classList.add('hidden-section');
        btnCert.classList.add('hidden-section');

        if (score === 100) {
            emoji.innerText = "🌟";
            msg.innerText = "完美！你已經完全掌握了這個階段的魔法。";
            playSnd('snd-win');
            if (currentPhase === 1) btnNext.classList.remove('hidden-section');
            else btnCert.classList.remove('hidden-section');
        } else {
            emoji.innerText = "🌧️";
            msg.innerText = "加油！還差一點點就能過關。森林長老規定要 100 分才能晉級喔！";
        }
    }

    function goToNextPhase() {
        currentPhase = 2;
        startPhase();
    }

    function showFinalCert() {
        document.getElementById('phase-result').classList.add('hidden-section');
        document.getElementById('quiz-success').classList.remove('hidden-section');
        playSnd('snd-win');
    }

    function updateProgress() {
        document.getElementById('progress-bar').style.width = (currentIdx / 5 * 100) + '%';
    }

    // 教學換算器
    function convertWeight(unit) {
        const t = document.getElementById('input-t');
        const kg = document.getElementById('input-kg');
        if (unit === 't') {
            const v = parseFloat(t.value);
            if (!isNaN(v)) {
                kg.value = (v * 1000);
                updateWeightUI(v);
            }
        } else {
            const v = parseFloat(kg.value);
            if (!isNaN(v)) {
                t.value = (v / 1000);
                updateWeightUI(v / 1000);
            }
        }
    }

    // 清空輸入欄位
    function clearWeightInput() {
        document.getElementById('input-t').value = '';
        document.getElementById('input-kg').value = '';
        updateWeightUI(5); // 重設為預設值
    }

    // 優化重量分級顯示（更新了多個全新動物與人類重量階層）
    function updateWeightUI(tVal) {
        const i = document.getElementById('weight-illustration');
        const d = document.getElementById('weight-desc');
        const kgVal = tVal * 1000;

        if (tVal <= 0.001) {
            i.innerText = '🥚';
            d.innerText = '好輕！這大約是一顆雞蛋的重量...';
        }
        else if (kgVal > 1 && kgVal < 50) {
            i.innerText = '🎒';
            d.innerText = `這重達 ${kgVal.toFixed(1)} 公斤，大約是幾個小學生書包的重量！`;
        }
