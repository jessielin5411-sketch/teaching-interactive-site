        // 將拼音與聲調切分，例如 "ㄔㄨㄢˊ" 切分為 { base: "ㄔㄨㄢ", tone: "ˊ" }
        function splitZhuyin(zhuyinStr) {
            const match = zhuyinStr.match(/^([ㄅㄆㄇㄈㄉㄊㄋㄌㄍㄎㄏㄐㄑㄒㄓㄔㄕㄖㄗㄘㄙㄚㄛㄜㄝㄞㄟㄠㄡㄢㄣㄤㄥㄦㄧㄨㄩ]+)([ˊˇˋ˙]?)$/);
            if (match) {
                return { base: match[1], tone: match[2] };
            }
            return { base: zhuyinStr, tone: '' };
        }

        // 解析包含 "字{ㄗˋ}" 格式的文本，並根據台灣標準注音排版渲染為 HTML
        // 【修正重點】正則表達式排除括號 `<` 與 `>`，避免 HTML 標籤（如 <br>）被吞進 ruby 中
        function parseZhuyin(text) {
            return text.replace(/([^\s{}()（）「」『』,，。！？<>]+)\{([^}]+)\}/g, (match, hanzi, zhuyin) => {
                const parts = splitZhuyin(zhuyin);
                let toneHtml = '';
                if (parts.tone === '˙') {
                    // 輕聲置於上方
                    toneHtml = `<span class="zhuyin-tone tone-light">${parts.tone}</span>`;
                } else if (parts.tone) {
                    // 其他聲調置於右側中間
                    toneHtml = `<span class="zhuyin-tone">${parts.tone}</span>`;
                }
                
                return `<ruby>${hanzi}<rt class="zhuyin-rt"><span class="zhuyin-base">${parts.base}</span>${toneHtml}</rt></ruby>`;
            });
        }

        // 切換注音顯示狀態
        function toggleZhuyin() {
            showZhuyin = !showZhuyin;
            if (showZhuyin) {
                gameContainer.classList.remove('no-zhuyin');
                zhuyinToggleBtn.innerHTML = "<span>注音：開 (ㄅㄆㄇ)</span>";
                zhuyinToggleBtn.className = "flex items-center gap-2 bg-[#d42426] text-white px-3 py-1 rounded-full text-xs font-bold hover:brightness-110 shadow-sm transition-all";
            } else {
                gameContainer.classList.add('no-zhuyin');
                zhuyinToggleBtn.innerHTML = "<span>注音：關 (國字)</span>";
                zhuyinToggleBtn.className = "flex items-center gap-2 bg-stone-400 text-white px-3 py-1 rounded-full text-xs font-bold hover:brightness-110 shadow-sm transition-all";
            }
        }

        function startGame() {
            startScreen.classList.add('hidden');
            quizScreen.classList.remove('hidden');
            resultScreen.classList.add('hidden');
            currentQuestion = 0;
            score = 0;
            showQuestion();
        }

        function showQuestion() {
            canAnswer = true;
            timeLeft = 15;
            feedback.style.opacity = '0';
            updateTimerDisplay();
            
            const q = questions[currentQuestion];
            document.getElementById('progress').innerText = `${currentQuestion + 1}/${questions.length}`;
            document.getElementById('score').innerText = score;
            
            // 渲染問題（直式注音）
            questionText.innerHTML = parseZhuyin(q.q);
            
            optionsContainer.innerHTML = '';
            q.o.forEach((opt, index) => {
                const btn = document.createElement('button');
                btn.className = "option-btn w-full text-left p-4 rounded-xl border-2 border-gray-100 hover:border-[#d4af37] hover:bg-orange-50 font-bold text-gray-700 transition-all flex justify-between items-center group leading-loose text-lg";
                // 渲染選項（直式注音）
                btn.innerHTML = `<span>${parseZhuyin(opt)}</span><span class="opacity-0 group-hover:opacity-100 text-xl">🧧</span>`;
                btn.onclick = () => handleAnswer(index);
                optionsContainer.appendChild(btn);
            });

            startTimer();
        }

        function startTimer() {
            clearInterval(timer);
            timer = setInterval(() => {
                timeLeft--;
                updateTimerDisplay();
                if (timeLeft <= 0) {
                    clearInterval(timer);
                    handleAnswer(-1); // 時間到
                }
            }, 1000);
        }

        function updateTimerDisplay() {
            timerCircle.innerText = timeLeft;
            if (timeLeft <= 5) {
                timerCircle.classList.add('text-red-500', 'border-red-500');
            } else {
                timerCircle.classList.remove('text-red-500', 'border-red-500');
            }
        }

        function handleAnswer(selectedIndex) {
            if (!canAnswer) return;
            canAnswer = false;
            clearInterval(timer);

            const correctIndex = questions[currentQuestion].a;
            const buttons = optionsContainer.querySelectorAll('button');

            if (selectedIndex === correctIndex) {
                // 答對
                score += (10 + timeLeft);
                buttons[selectedIndex].classList.add('bg-green-100', 'border-green-500', 'text-green-800');
                showFeedback(parseZhuyin("答{ㄉㄚˊ}對{ㄉㄨㄟˋ}了{ㄌㄜ˙}！恭{ㄍㄨㄥ}喜{ㄒㄧˇ}發{ㄈㄚ}財{ㄘㄞˊ} 🧧"), "text-green-600");
            } else {
                // 答錯
                if (selectedIndex !== -1) {
                    buttons[selectedIndex].classList.add('bg-red-100', 'border-red-500', 'text-red-800');
                }
                buttons[correctIndex].classList.add('bg-green-50', 'border-green-300');
                showFeedback(selectedIndex === -1 ? parseZhuyin("時{ㄕˊ}間{ㄐㄧㄢ}到{ㄉㄠˋ}！正{ㄓˋ}確{ㄑㄩㄝˋ}答{ㄉㄚˊ}案{ㄢˋ}是{ㄕˋ}...") : parseZhuyin("哎{ㄞ}呀{ㄧㄚ}，答{ㄉㄚˊ}錯{ㄘㄨㄛˋ}了{ㄌㄜ˙}！"), "text-red-600");
            }

            setTimeout(() => {
                currentQuestion++;
                if (currentQuestion < questions.length) {
                    showQuestion();
                } else {
                    showResult();
                }
            }, 2500); // 提供充足時間閱讀直式注音回饋
        }

        function showFeedback(htmlText, colorClass) {
            feedback.innerHTML = htmlText;
            feedback.className = `h-12 mt-4 flex items-center justify-center font-bold transition-all opacity-100 ${colorClass}`;
        }

        function showResult() {
            quizScreen.classList.add('hidden');
            resultScreen.classList.remove('hidden');
            document.getElementById('final-score').innerText = score;
            
            let comment = "";
            let icon = "";
            if (score >= 200) {
                comment = "你{ㄋㄧˇ}簡{ㄐㄧㄢˇ}直{ㄓˊ}是{ㄕˋ}春{ㄔㄨㄣ}節{ㄐㄧㄝˊ}文{ㄨㄣˊ}化{ㄏㄨㄚˋ}大{ㄉㄚˋ}師{ㄕ}！全{ㄑㄩㄢˊ}家{ㄐㄧㄚ}人{ㄖㄣˊ}都{ㄉㄡ}要{ㄧㄠˋ}聽{ㄊㄧㄥ}你{ㄋㄧˇ}的{ㄉㄜ˙}！";
                icon = "👑";
            } else if (score >= 150) {
                comment = "非{ㄈㄟ}常{ㄔㄤˊ}厲{ㄌㄧˋ}害{ㄏㄞˋ}！今{ㄐㄧㄣ}年{ㄋㄧˊ}的{ㄉㄜ˙}紅{ㄏㄨㄥˊ}包{ㄅㄠ}一{ㄧ}定{ㄉㄧˋ}特{ㄊㄜˋ}別{ㄅㄧㄝˊ}厚{ㄏㄡˋ}！";
                icon = "💰";
            } else if (score >= 100) {
                comment = "表{ㄅㄧㄠˇ}現{ㄒㄧㄢˋ}不{ㄅㄨˊ}錯{ㄘㄨㄛˋ}喔{ㄛ}！對{ㄉㄨˋ}傳{ㄔㄨㄢˊ}統{ㄊㄨㄥˇ}文{ㄨㄣˊ}化{ㄏㄨㄚˋ}很{ㄏㄣˇ}有{ㄧㄡˇ}研{ㄧㄢˊ}究{ㄐㄧㄡˋ}。";
                icon = "🏮";
            } else {
                comment = "再{ㄗㄞˋ}接{ㄐㄧㄝ}再{ㄗㄞˋ}厲{ㄌㄧˋ}，多{ㄉㄨㄛ}跟{ㄍㄣ}長{ㄓㄤˇ}輩{ㄅㄟˋ}聊{ㄌㄧㄠˊ}天{ㄊㄧㄢ}學{ㄒㄩㄝˊ}習{ㄒㄧˊ}新{ㄒㄧㄣ}春{ㄔㄨㄣ}冷{ㄌㄥˇ}知{ㄓ}識{ㄕ}！";
                icon = "🍊";
            }
            document.getElementById('result-comment').innerHTML = parseZhuyin(comment);
            document.getElementById('result-icon').innerText = icon;
        }

        function resetGame() {
            startGame();
        }
