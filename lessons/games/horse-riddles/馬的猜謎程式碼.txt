    createApp({
        setup() {
            const gameState = ref('menu');
            const mode = ref('');
            const activePool = ref([]);
            const currentIdx = ref(0);
            const visibleCluesCount = ref(1);
            const userAnswer = ref('');
            const totalScore = ref(0);
            const correctCount = ref(0);
            const feedback = ref(null);
            const timer = ref(30);
            const shuffledOptions = ref([]);
            let timerInterval = null;

            const currentQuestion = computed(() => activePool.value[currentIdx.value]);
            const visibleClues = computed(() => currentQuestion.value?.clues.slice(0, visibleCluesCount.value) || []);
            const timerClass = computed(() => {
                if (timer.value <= 5) return 'bg-red-500';
                if (timer.value <= 15) return 'bg-yellow-500';
                return 'bg-green-500';
            });

            const performanceRating = computed(() => {
                const ratio = correctCount.value / activePool.value.length;
                if (ratio === 1) return "🥇 絕世神馬 (滿分！)";
                if (ratio >= 0.8) return "🌟 萬馬之王 (極其優秀)";
                return "💪 潛力黑馬 (加油再試試)";
            });

            const selectMode = (m) => {
                mode.value = m;
                startCurrentMode();
            };

            const startCurrentMode = () => {
                SoundManager.init();
                if (mode.value === 'idioms') {
                    // 成語隨機抽 10 題
                    activePool.value = [...horseIdiomsGame].sort(() => Math.random() - 0.5).slice(0, 10);
                } else {
                    // 名稱篇使用更新後的 8 題
                    activePool.value = [...horseNameTrivia];
                }
                currentIdx.value = 0;
                totalScore.value = 0;
                correctCount.value = 0;
                gameState.value = 'playing';
                initQuestion();
            };

            const initQuestion = () => {
                userAnswer.value = '';
                visibleCluesCount.value = 1;
                shuffledOptions.value = [...currentQuestion.value.options].sort(() => Math.random() - 0.5);
                startTimer();
            };

            const startTimer = () => {
                timer.value = 30;
                clearInterval(timerInterval);
                timerInterval = setInterval(() => {
                    timer.value--;
                    if (timer.value <= 0) { clearInterval(timerInterval); checkAnswer(); }
                }, 1000);
            };

            const showNextClue = () => {
                if (visibleCluesCount.value < currentQuestion.value.clues.length) {
                    visibleCluesCount.value++;
                    totalScore.value = Math.max(0, totalScore.value - 10); // 增加線索扣 10 分
                    SoundManager.play(800, 'sine', 0.05);
                }
            };

            const selectOption = (opt) => { userAnswer.value = opt; checkAnswer(); };

            const checkAnswer = () => {
                if (userAnswer.value.trim() === currentQuestion.value.answer) {
                    clearInterval(timerInterval);
                    SoundManager.play(600, 'sine', 0.1);
                    feedback.value = 'correct';
                    correctCount.value++;
                    const bonus = timer.value > 20 ? 50 : 30;
                    totalScore.value += Math.max(10, bonus + (currentQuestion.value.clues.length - visibleCluesCount.value) * 10);
                    confetti({ particleCount: 100, spread: 70, origin: { y: 0.6 } });
                } else if (userAnswer.value !== '') {
                    totalScore.value = Math.max(0, totalScore.value - 20); // 答錯扣 20 分
                    SoundManager.play(150, 'square', 0.2);
                    feedback.value = 'wrong';
                }
            };

            const nextQuestion = () => {
                feedback.value = null;
                if (currentIdx.value < activePool.value.length - 1) {
                    currentIdx.value++;
                    initQuestion();
                } else {
                    gameState.value = 'end';
                }
            };

            const exitGame = () => {
                clearInterval(timerInterval);
                gameState.value = 'menu';
            };

            return {
                gameState, mode, activePool, currentIdx, visibleCluesCount, userAnswer, totalScore, correctCount, feedback, timer, shuffledOptions,
                currentQuestion, visibleClues, timerClass, performanceRating,
                selectMode, startCurrentMode, exitGame, showNextClue, selectOption, checkAnswer, nextQuestion
            };
        }
    }).mount('#app');
</script>
